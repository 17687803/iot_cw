197 num_count++; //printf("%d numbers\r\n", num_count);
198 ss.str("");
199 ss.clear();
200 /* if fininshed, calculate answer & store in char array */
201 } else if (extracted_str == "FINISHED") { //printf("calculating..\r\n");
202 math = 0;
203 num_count++; //printf("%d numbers\r\n", num_count);
204 for (int i = 0; i < num_count; i++) {
205 if (i == 0) {

207 } else {
208 if (ALU == 0) {

216 }
217 } //printf("num: %f\r\n", answer);

219 if (floor(answer) == answer) {
220 sprintf(num_str, "The answer is %d", (int)answer);
